MikroTik Setup Script
.rsc 
# SpotPay MikroTik Setup — {{ location.site_name }}
# Paste into MikroTik Terminal (New Terminal in Winbox)

# 1. Walled Garden — allow SpotPay server
/ip hotspot walled-garden ip
add action=accept comment="SpotPay" dst-host={{ request.get_host }}

/ip hotspot walled-garden
add action=allow comment="SpotPay API" dst-host={{ request.get_host }}

# 2. Download portal files
/tool fetch url="{{ site_url }}/api/portal/{{ location.uuid }}/download/" \
    dst-path="hotspot/hotspot-spotpay.zip"

# 3. Extract (RouterOS 7.x only)
/file extract hotspot/hotspot-spotpay.zip to=hotspot

# 4. DNS name must be set to: {{ location.hotspot_dns|default:"hot.spot" }}
#    IP -> Hotspot -> Server Profiles -> DNS Name

:put "SpotPay setup complete."
