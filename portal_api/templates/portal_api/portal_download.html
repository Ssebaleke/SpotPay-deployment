# =======================================================
# SpotPay MikroTik Setup Script
# Location : {{ location.site_name }}
# Paste this entire block into MikroTik Terminal
# =======================================================

# --- 1. Walled Garden: allow SpotPay server IP ---
/ip hotspot walled-garden ip
add action=accept comment="SpotPay Server" dst-host={{ site_url|slice:"7:" }}

/ip hotspot walled-garden
add action=allow comment="SpotPay API" dst-host={{ site_url|slice:"7:" }}

# --- 2. Download portal ZIP from SpotPay ---
/tool fetch url="{{ site_url }}/api/portal/{{ location.uuid }}/download/" \
    dst-path="hotspot/hotspot-spotpay.zip"

# --- 3. RouterOS 7.x: extract ZIP in place ---
# (Skip this on RouterOS 6.x — extract on PC and upload via Winbox Files)
/file extract hotspot/hotspot-spotpay.zip to=hotspot

# --- 4. Confirm DNS name in hotspot profile ---
# IP -> Hotspot -> Server Profiles -> DNS Name must be:
#   {{ location.hotspot_dns|default:"hot.spot" }}

:log info "SpotPay setup complete: {{ location.site_name }}"
:put "Done. Reload hotspot to apply."
The script adds walled garden rules so clients can reach SpotPay before paying, then downloads and extracts your portal files directly onto the router.
